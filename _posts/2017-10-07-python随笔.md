# 生成器

        >>> def odd():
        ...     print 'step 1'
        ...     yield 1
        ...     print 'step 2'
        ...     yield 3
        ...     print 'step 3'
        ...     yield 5        
        >>> o = odd()
        >>> o.next()
        step 1
        1
        >>> o.next()
        step 2
        3
        >>> o.next()
        step 3
        5

# 文件读取

        def read_file(path):
            s = set([])
            f = open(path, 'r')
            for line in f.readlines():
                s.add(line.strip())
            return s
