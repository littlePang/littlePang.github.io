# 导入与导出
        导出表结构：mysqldump -u用户名 -p密码 -d 数据库名 > db.sql
        导出表结构和数据：mysqldump -u用户名 -p密码 数据库名 > db.sql
        导入sql文件：source /home/abc/db.sql;
        导入sql文件：mysql -u用户名 -p密码 数据库名 < db.sql
